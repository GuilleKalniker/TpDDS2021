@startuml tpa

enum TipoDoc {
    - DNI
    - LC
    - LE
    - CI
}

enum TipoMascota {
    - PERRO
    - GATO
}

enum Sexo {
    - MASCULINO
    - FEMENINO
}

class Foto {
    - url: String
}

class Contacto {
    - nombre: String
    - apellido: String
    - telefono: Integer
    - email: String
}

class Duenio {
    - nombre: String
    - apellido: String
    - fechaNacimiento: Date
    - tipoDocumento: TipoDNI
    - nroDocumento: Integer
    - contacto: List<Contacto>
    - mascotas: List<MascotaRegistrada>

    +registrarMascota()
}

class MascotaRegistrada {
    - tipo: TipoMascota
    - nombre: String
    - apodo: String
    - edad: Integer
    - sexo: Sexo
    - descripcionFisica: String
    - fotos: List<Foto>
    - qr: Integer
    - duenio: Duenio
}

class EstadoMascotaPerdida {
    - rescatistaDeEncuentro: Rescatista
    - descripcionEstado: String
    - lugarEncuentro: String
    - fechaEncuentro: Date
    - qrMascotaPerdida: Integer
}

class Rescatista {
    - nombre: String
    - apellido: String
    - fechaDeNacimiento: Date
    - tipoDocumento: TipoDocumento
    - nroDocumento: Integer
    - direccion: String
    - contacto: List<Contacto>

    + notificarMascotaEncontrada(mascota)
}

class Administrador {
    + agregarCaracteristica()
    + agregarValorACaracteristica()
}

class CentroDeRescate {
    - mascotasRegistradas: List<MascotaRegistrada>
    - estadosMascotasPerdidas: List<EstadoMascotaPerdida>
    - caracteristicas: Caracteristicas

    + otorgarQR()
    + notificarMascotaEncontrada(mascota)
    + agregarMascotaRegistrada(mascota)
    + agregarEstadoMascotaPerdida()
}

class Caracteristicas {
    - DiccionarioCaracteristicas: HashMap
}

TipoDoc -> Duenio
TipoDoc -> Rescatista
Contacto "*" -- "*" Duenio
Contacto "*" -- "*" Rescatista
TipoMascota -> MascotaRegistrada
Sexo -> MascotaRegistrada
Foto "*" -- "1" MascotaRegistrada
Duenio "1" -- "*" MascotaRegistrada
Caracteristicas -> CentroDeRescate
EstadoMascotaPerdida -> CentroDeRescate
MascotaRegistrada -> CentroDeRescate
Rescatista -> EstadoMascotaPerdida
Administrador ..> CentroDeRescate
Administrador ..> Caracteristicas

@enduml
